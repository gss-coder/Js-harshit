### Loops 

#### 1. While loop
// While loop - [0-9]

```js
// While loop till 9 
while(i<=9)
{
    // Log the o/p
    console.log(i);
    
    // Incr the counter
    i++;
}

// O/p the ans
console.log(`current value of i is ${i}`);
console.log("hello");
```

#### 2. For loop

```js
// Example 1
// Intro to for loop 
// Iterate on the loop

for(let i = 0;i<=9;i++)
{
    console.log(i);
}

// O/p the ans
console.log("value of i is ",i);

```


```js
// Example 2

// Declare the vars
let total = 0;
let num = 100;

// Iterate on the loop
for(let i = 1; i<=num; i++)
{
    total = total + i;
}

// O/p the ans
console.log(total);

```
